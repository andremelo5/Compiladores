package main

import "fmt"

func main() {
    // Illegal characters in string literal
    str1 := "Hello\k World"
    
    // Illegal character in rune literal
    r1 := '\k'
    
    // Illegal Unicode escape sequence
    str2 := "\U0000FFFFF"
    
    // Illegal octal escape sequence
    str3 := "\400"
    
    fmt.Println(str1, r1, str2, str3)
}
